```.py
from mylibraly import numberonly

a = input("input strings that want to shift")
m = int(numberonly(input("how many letters do you wanna shift?")))
b = ""
for i in a:
    if 97 <= ord(i) <= 122:
        b += chr((ord(i) - 97 + m) % 26 + 97)
    elif 65 <= ord(i) <= 90:
        b += chr((ord(i) - 65 + m) % 26 + 65)
    else:
        b += i
print(b)
```
