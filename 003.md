elif DNA == "G":
    print("C")
elif DNA == "T":
    print("A")
elif DNA == "C":
    print("G")

    print("You write wrong thing.")

        print("T")
    elif b[i] == "G":
        print("C")
    elif b[i] == "T":
        print("A")
